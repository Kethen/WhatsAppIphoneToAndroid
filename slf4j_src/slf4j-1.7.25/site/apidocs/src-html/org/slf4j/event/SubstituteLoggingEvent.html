001package org.slf4j.event;
002
003import org.slf4j.Marker;
004import org.slf4j.helpers.SubstituteLogger;
005
006public class SubstituteLoggingEvent implements LoggingEvent {
007
008    Level level;
009    Marker marker;
010    String loggerName;
011    SubstituteLogger logger;
012    String threadName;
013    String message;
014    Object[] argArray;
015    long timeStamp;
016    Throwable throwable;
017
018    public Level getLevel() {
019        return level;
020    }
021
022    public void setLevel(Level level) {
023        this.level = level;
024    }
025
026    public Marker getMarker() {
027        return marker;
028    }
029
030    public void setMarker(Marker marker) {
031        this.marker = marker;
032    }
033
034    public String getLoggerName() {
035        return loggerName;
036    }
037
038    public void setLoggerName(String loggerName) {
039        this.loggerName = loggerName;
040    }
041
042    public SubstituteLogger getLogger() {
043        return logger;
044    }
045
046    public void setLogger(SubstituteLogger logger) {
047        this.logger = logger;
048    }
049
050    public String getMessage() {
051        return message;
052    }
053
054    public void setMessage(String message) {
055        this.message = message;
056    }
057
058    public Object[] getArgumentArray() {
059        return argArray;
060    }
061
062    public void setArgumentArray(Object[] argArray) {
063        this.argArray = argArray;
064    }
065
066    public long getTimeStamp() {
067        return timeStamp;
068    }
069
070    public void setTimeStamp(long timeStamp) {
071        this.timeStamp = timeStamp;
072    }
073
074    public String getThreadName() {
075        return threadName;
076    }
077
078    public void setThreadName(String threadName) {
079        this.threadName = threadName;
080    }
081
082    public Throwable getThrowable() {
083        return throwable;
084    }
085
086    public void setThrowable(Throwable throwable) {
087        this.throwable = throwable;
088    }
089}
