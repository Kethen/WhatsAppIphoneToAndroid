025package org.slf4j.impl;
026
027import org.slf4j.IMarkerFactory;
028import org.slf4j.MarkerFactory;
029import org.slf4j.helpers.BasicMarkerFactory;
030import org.slf4j.spi.MarkerFactoryBinder;
031
032/**
033 * 
034 * The binding of {@link MarkerFactory} class with an actual instance of 
035 * {@link IMarkerFactory} is performed using information returned by this class. 
036 * 
037 * This class is meant to provide a *dummy* StaticMarkerBinder to the slf4j-api module. 
038 * Real implementations are found in  each SLF4J binding project, e.g. slf4j-nop, 
039 * slf4j-simple, slf4j-log4j12 etc.
040 * 
041 * @author Ceki G&uuml;lc&uuml;
042 */
043public class StaticMarkerBinder implements MarkerFactoryBinder {
044
045    /**
046     * The unique instance of this class.
047     */
048    public static final StaticMarkerBinder SINGLETON = new StaticMarkerBinder();
049
050    private StaticMarkerBinder() {
051        throw new UnsupportedOperationException("This code should never make it into the jar");
052    }
053
054    /**
055     * Return the singleton of this class.
056     * 
057     * @return the StaticMarkerBinder singleton
058     * @since 1.7.14
059     */
060    public static StaticMarkerBinder getSingleton() {
061        return SINGLETON;
062    }
063
064    /**
065     * Currently this method always returns an instance of 
066     * {@link BasicMarkerFactory}.
067     */
068    public IMarkerFactory getMarkerFactory() {
069        throw new UnsupportedOperationException("This code should never make it into the jar");
070    }
071
072    /**
073     * Currently, this method returns the class name of
074     * {@link BasicMarkerFactory}.
075     */
076    public String getMarkerFactoryClassStr() {
077        throw new UnsupportedOperationException("This code should never make it into the jar");
078    }
079
080}
