017package org.apache.log4j.spi;
018
019import org.apache.log4j.spi.LoggerRepository;
020import java.net.URL;
021
022/**
023   Implemented by classes capable of configuring log4j using a URL.
024
025   @since 1.0
026   @author Anders Kristensen
027 */
028public interface Configurator {
029
030    /**
031       Special level value signifying inherited behaviour. The current
032       value of this string constant is <b>inherited</b>. {@link #NULL}
033       is a synonym.  */
034    public static final String INHERITED = "inherited";
035
036    /**
037       Special level signifying inherited behaviour, same as {@link
038       #INHERITED}. The current value of this string constant is
039       <b>null</b>. */
040    public static final String NULL = "null";
041
042    /**
043       Interpret a resource pointed by a URL and set up log4j accordingly.
044
045       The configuration is done relative to the <code>hierarchy</code>
046       parameter.
047
048       @param url The URL to parse
049       @param repository The hierarchy to operation upon.
050     */
051    void doConfigure(URL url, LoggerRepository repository);
052}
