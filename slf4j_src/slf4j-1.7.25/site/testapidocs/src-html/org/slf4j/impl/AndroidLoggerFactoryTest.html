025package org.slf4j.impl;
026
027import org.junit.Test;
028
029import static org.junit.Assert.assertEquals;
030
031public class AndroidLoggerFactoryTest {
032    @Test
033    public void shortLoggerNames() {
034        assertEquals("o.test.p.TestClass", AndroidLoggerFactory.loggerNameToTag("o.test.p.TestClass"));
035        assertEquals("ex.test.TestClass", AndroidLoggerFactory.loggerNameToTag("ex.test.TestClass"));
036        assertEquals("MyClass", AndroidLoggerFactory.loggerNameToTag("MyClass"));
037    }
038
039    @Test
040    public void emptyLoggerNames() {
041        assertEquals(AndroidLoggerFactory.ANONYMOUS_TAG, AndroidLoggerFactory.loggerNameToTag(null));
042        assertEquals("", AndroidLoggerFactory.loggerNameToTag(""));
043    }
044
045    @Test
046    public void simpleLoggerName() {
047        assertEquals("o*.t*.p*.TestClass", AndroidLoggerFactory.loggerNameToTag("org.test.package.TestClass"));
048    }
049
050    @Test
051    public void loggerNameWithOneCharPackage() {
052        assertEquals("o.t*.p*.p*.TestClass", AndroidLoggerFactory.loggerNameToTag("o.test.project.package.TestClass"));
053        assertEquals("o.t*.p*.p.TestClass", AndroidLoggerFactory.loggerNameToTag("o.test.project.p.TestClass"));
054    }
055
056    @Test
057    public void longLoggerName() {
058        assertEquals("AndroidLoggerFactory", AndroidLoggerFactory.loggerNameToTag("org.slf4j.impl.AndroidLoggerFactory"));
059    }
060
061    @Test
062    public void veryLongLoggerName() {
063        assertEquals("*meAndShouldBeTruncated", AndroidLoggerFactory.loggerNameToTag("IAmAVeryLongLoggerNameAndShouldBeTruncated"));
064    }
065
066    @Test
067    public void oneWordLoggerName() {
068        assertEquals("TestClass", AndroidLoggerFactory.loggerNameToTag("TestClass"));
069    }
070
071    @Test
072    public void weirdLoggerNames() {
073        assertEquals("WeirdLoggerName.", AndroidLoggerFactory.loggerNameToTag("WeirdLoggerName."));
074        assertEquals(".WeirdLoggerName", AndroidLoggerFactory.loggerNameToTag(".WeirdLoggerName"));
075        assertEquals(".WeirdLoggerName.", AndroidLoggerFactory.loggerNameToTag(".WeirdLoggerName."));
076        assertEquals(".", AndroidLoggerFactory.loggerNameToTag("."));
077        assertEquals("..", AndroidLoggerFactory.loggerNameToTag(".."));
078    }
079}
