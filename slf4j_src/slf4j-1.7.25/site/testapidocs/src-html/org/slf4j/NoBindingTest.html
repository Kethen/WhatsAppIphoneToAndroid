025package org.slf4j;
026
027import static org.junit.Assert.assertNull;
028import static org.junit.Assert.assertTrue;
029
030import java.util.Random;
031
032import org.junit.Test;
033import org.slf4j.helpers.BasicMarker;
034import org.slf4j.helpers.NOPLogger;
035
036public class NoBindingTest {
037
038    int diff = new Random().nextInt(10000);
039
040    @Test
041    public void testLogger() {
042        Logger logger = LoggerFactory.getLogger(NoBindingTest.class);
043        logger.debug("hello" + diff);
044        assertTrue(logger instanceof NOPLogger);
045    }
046
047    @Test
048    public void testMDC() {
049        MDC.put("k" + diff, "v");
050        assertNull(MDC.get("k"));
051    }
052
053    @Test
054    public void testMarker() {
055        Marker m = MarkerFactory.getMarker("a" + diff);
056        assertTrue(m instanceof BasicMarker);
057    }
058}
