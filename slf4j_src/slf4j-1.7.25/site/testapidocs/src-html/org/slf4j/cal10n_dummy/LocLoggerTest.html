025package org.slf4j.cal10n_dummy;
026
027import static org.junit.Assert.assertEquals;
028
029import java.util.Locale;
030
031import org.apache.log4j.spi.LoggingEvent;
032import org.junit.Before;
033import org.junit.Test;
034import org.slf4j.cal10n.LocLogger;
035import org.slf4j.cal10n.LocLoggerFactory;
036import org.slf4j.dummyExt.ListAppender;
037
038import ch.qos.cal10n.IMessageConveyor;
039import ch.qos.cal10n.MessageConveyor;
040
041public class LocLoggerTest {
042
043    ListAppender listAppender;
044    org.apache.log4j.Logger log4jRoot;
045
046    IMessageConveyor imc = new MessageConveyor(Locale.UK);
047    LocLoggerFactory llFactory_uk = new LocLoggerFactory(imc);
048
049    final static String EXPECTED_FILE_NAME = "LocLoggerTest.java";
050
051    @Before
052    public void setUp() throws Exception {
053        // start from a clean slate for each test
054
055        listAppender = new ListAppender();
056        listAppender.extractLocationInfo = true;
057        log4jRoot = org.apache.log4j.Logger.getRootLogger();
058        log4jRoot.addAppender(listAppender);
059        log4jRoot.setLevel(org.apache.log4j.Level.TRACE);
060    }
061
062    void verify(LoggingEvent le, String expectedMsg) {
063        assertEquals(expectedMsg, le.getMessage());
064        assertEquals(EXPECTED_FILE_NAME, le.getLocationInformation().getFileName());
065    }
066
067    @Test
068    public void testSmoke() {
069        LocLogger locLogger = llFactory_uk.getLocLogger(this.getClass());
070        locLogger.info(Months.JAN);
071        verify((LoggingEvent) listAppender.list.get(0), "January");
072
073    }
074}
