25  package org.slf4j;
26  
27  /**
28   * Implementations of this interface are used to manufacture {@link Marker}
29   * instances.
30   * 
31   * <p>See the section <a href="http://slf4j.org/faq.html#3">Implementing 
32   * the SLF4J API</a> in the FAQ for details on how to make your logging 
33   * system conform to SLF4J.
34   * 
35   * @author Ceki G&uuml;lc&uuml;
36   */
37  public interface IMarkerFactory {
38  
39      /**
40       * Manufacture a {@link Marker} instance by name. If the instance has been 
41       * created earlier, return the previously created instance. 
42       * 
43       * <p>Null name values are not allowed.
44       *
45       * @param name the name of the marker to be created, null value is
46       * not allowed.
47       *
48       * @return a Marker instance
49       */
50      Marker getMarker(String name);
51  
52      /**
53       * Checks if the marker with the name already exists. If name is null, then false 
54       * is returned.
55       *  
56       * @param name logger name to check for
57       * @return true id the marker exists, false otherwise. 
58       */
59      boolean exists(String name);
60  
61      /**
62       * Detach an existing marker.
63       * <p>
64       * Note that after a marker is detached, there might still be "dangling" references
65       * to the detached marker.
66       * 
67       * 
68       * @param name The name of the marker to detach
69       * @return whether the marker  could be detached or not
70       */
71      boolean detachMarker(String name);
72  
73      /**
74       * Create a marker which is detached (even at birth) from this IMarkerFactory.
75       * 
76       * @param name marker name
77       * @return a dangling marker
78       * @since 1.5.1
79       */
80      Marker getDetachedMarker(String name);
81  }
