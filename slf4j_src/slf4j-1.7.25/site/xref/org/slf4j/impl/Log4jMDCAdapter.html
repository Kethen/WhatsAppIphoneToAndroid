25  package org.slf4j.impl;

27  import java.util.HashMap;
28  import java.util.Iterator;
29  import java.util.Map;
30  
31  import org.apache.log4j.MDCFriend;
32  import org.slf4j.spi.MDCAdapter;
33  
34  public class Log4jMDCAdapter implements MDCAdapter {
35  
36  	static {
37  		if (VersionUtil.getJavaMajorVersion() >= 9) {
38  			MDCFriend.fixForJava9();
39  		}
40  	}
41  
42  	public void clear() {
43  		@SuppressWarnings("rawtypes")
44  		Map map = org.apache.log4j.MDC.getContext();
45  		if (map != null) {
46  			map.clear();
47  		}
48  	}
49  
50  	public String get(String key) {
51  		return (String) org.apache.log4j.MDC.get(key);
52  	}
53  
54  	/**
55  	 * Put a context value (the <code>val</code> parameter) as identified with
56  	 * the <code>key</code> parameter into the current thread's context map. The
57  	 * <code>key</code> parameter cannot be null. Log4j does <em>not</em>
58  	 * support null for the <code>val</code> parameter.
59  	 * 
60  	 * <p>
61  	 * This method delegates all work to log4j's MDC.
62  	 * 
63  	 * @throws IllegalArgumentException
64  	 *             in case the "key" or <b>"val"</b> parameter is null
65  	 */
66  	public void put(String key, String val) {
67  		org.apache.log4j.MDC.put(key, val);
68  	}
69  
70  	public void remove(String key) {
71  		org.apache.log4j.MDC.remove(key);
72  	}
73  
74  	@SuppressWarnings({ "rawtypes", "unchecked" })
75  	public Map getCopyOfContextMap() {
76  		Map old = org.apache.log4j.MDC.getContext();
77  		if (old != null) {
78  			return new HashMap(old);
79  		} else {
80  			return null;
81  		}
82  	}
83  
84  	@SuppressWarnings({ "rawtypes", "unchecked" })
85  	public void setContextMap(Map contextMap) {
86  		Map old = org.apache.log4j.MDC.getContext();
87  		if (old == null) {
88  			Iterator entrySetIterator = contextMap.entrySet().iterator();
89  			while (entrySetIterator.hasNext()) {
90  				Map.Entry mapEntry = (Map.Entry) entrySetIterator.next();
91  				org.apache.log4j.MDC.put((String) mapEntry.getKey(), mapEntry.getValue());
92  			}
93  		} else {
94  			old.clear();
95  			old.putAll(contextMap);
96  		}
97  	}
98  }
