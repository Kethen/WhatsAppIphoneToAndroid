25  package org.slf4j.migrator.line;
26  
27  import java.util.ArrayList;
28  import java.util.Iterator;
29  import java.util.regex.Pattern;
30  
31  /**
32   * This class represents JCL to SLF4J conversion rules
33   * 
34   * @author Jean-Noel Charpin
35   * 
36   */
37  public class JCLRuleSet implements RuleSet {
38  
39      private ArrayList<ConversionRule> conversionRuleList;
40  
41      public JCLRuleSet() {
42          // matching : import org.apache.commons.logging.LogFactory;
43          SingleConversionRule cr0 = new SingleConversionRule(Pattern.compile("import\\s*+org.apache.commons.logging.LogFactory;"),
44                          "import org.slf4j.LoggerFactory;");
45  
46          // matching : import org.apache.commons.logging.Log;
47          SingleConversionRule cr1 = new SingleConversionRule(Pattern.compile("import\\s*+org.apache.commons.logging.Log;"), "import org.slf4j.Logger;");
48  
49          SingleConversionRule cr2 = new SingleConversionRule(Pattern.compile("(\\sLog\\b)"), " Logger");
50  
51          SingleConversionRule cr3 = new SingleConversionRule(Pattern.compile("(^Log\\b)"), "Logger");
52  
53          SingleConversionRule cr4 = new SingleConversionRule(Pattern.compile("LogFactory.getFactory\\(\\).getInstance\\("), "LoggerFactory.getLogger(");
54  
55          SingleConversionRule cr5 = new SingleConversionRule(Pattern.compile("LogFactory.getLog\\("), "LoggerFactory.getLogger(");
56  
57          conversionRuleList = new ArrayList<ConversionRule>();
58          conversionRuleList.add(cr0);
59          conversionRuleList.add(cr1);
60          conversionRuleList.add(cr2);
61          conversionRuleList.add(cr3);
62          conversionRuleList.add(cr4);
63          conversionRuleList.add(cr5);
64      }
65  
66      public Iterator<ConversionRule> iterator() {
67          return conversionRuleList.iterator();
68      }
69  }
