25  package org.slf4j.migrator.line;
26  
27  import java.util.ArrayList;
28  import java.util.Iterator;
29  import java.util.regex.Pattern;
30  
31  import org.slf4j.migrator.line.ConversionRule;
32  import org.slf4j.migrator.line.MultiGroupConversionRule;
33  import org.slf4j.migrator.line.RuleSet;
34  import org.slf4j.migrator.line.SingleConversionRule;
35  
36  class TrivialMatcher implements RuleSet {
37  
38      private ArrayList<ConversionRule> conversionRuleList;
39  
40      public TrivialMatcher() {
41          // simple rule no capturing group is defined, we use default capturing group which is group zero
42          SingleConversionRule cr = new SingleConversionRule(Pattern.compile("import org.slf4j.converter"), "simple replacement with an unique capturing group");
43  
44          // we define 4 differents capturing groups
45          MultiGroupConversionRule cr1 = new MultiGroupConversionRule(Pattern.compile("(first group)( second group)( third group)( 4th group)"));
46          // group zero is ignored during treatment
47          // replacement for the first
48          cr1.addReplacement(1, "1st group");
49          // no replacement for the second group it will remains the same
50          // empty string for the third group it will be deleted
51          cr1.addReplacement(3, "");
52          // no replacement for the third group it will remains the same
53  
54          conversionRuleList = new ArrayList<ConversionRule>();
55          conversionRuleList.add(cr);
56          conversionRuleList.add(cr1);
57      }
58  
59      public Iterator<ConversionRule> iterator() {
60          return conversionRuleList.iterator();
61      }
62  
63  }
